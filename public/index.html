Agent Snippet

Copy this into your agent's startup script:

BASE="http://localhost:4444/api"
PORT=3000
AGENT="my-agent"
REASON="Dev server"

# 1. Check if port is available
STATUS=$(curl -s "$BASE/ports/check/$PORT")
AVAILABLE=$(echo $STATUS | python -c \
  "import sys,json; print(json.load(sys.stdin)['available'])")

if [ "$AVAILABLE" = "True" ]; then
  # 2. Register the port
  curl -s -X POST "$BASE/ports/register" \
    -H "Content-Type: application/json" \
    -d "{\"port\":$PORT,\"agent\":\"$AGENT\",\"reason\":\"$REASON\"}"
  echo "Port $PORT registered."
else
  echo "Port $PORT is taken. Try another."
  # Optionally: GET $BASE/suggest to find a free one
fi
